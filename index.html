export const runtime = 'edge'; export async function POST(req) { try { const { messages, model } = await req.json(); const GROQ_API_KEY = process.env.GROQ_API_KEY; if (!GROQ_API_KEY) { return new Response(JSON.stringify({ error: 'Server tidak dikonfigurasi.' }), { status: 500 }); } const response = await fetch('https://api.groq.com/openai/v1/chat/completions', { method: 'POST', headers: { 'Authorization': `Bearer ${GROQ_API_KEY}`, 'Content-Type': 'application/json', }, body: JSON.stringify({ model: model || 'llama-3.1-8b-instant', messages: [ { role: 'system', content: 'Kamu adalah Takon AI, asisten berbahasa Indonesia yang ramah dan membantu.' }, ...messages ], stream: true, temperature: 0.7, max_tokens: 4096, }), }); return new Response(response.body, { headers: { 'Content-Type': 'text/event-stream', 'Cache-Control': 'no-cache', 'Connection': 'keep-alive', }, }); } catch (error) { return new Response(JSON.stringify({ error: error.message }), { status: 500 }); } }
